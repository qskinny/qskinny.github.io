#ifndef QSK_STANDARD_SYMBOL_H

#include <qmetatype.h>

{
    Q_GADGET

  public:
    enum Type
    {
        NoSymbol = -1,

        Ok,
        Cancel,

        Information,
        Warning,
        Critical,
        Question,

        SymbolTypeCount
    };

};
